CDN Usage

Quick Start

Get started immediately with our CDN bundles. Choose between the complete bundle (all plugins) or framework-only bundle.

<!-- Complete Bundle: Framework + All Plugins -->
<script src="https://cdn.jsdelivr.net/npm/@dataprism/plugins@latest/dist/bundles/dataprism-plugins.min.js"></script>

<script>
  // All plugins are immediately available
  const manager = await DataPrismPlugins.createPluginManager();
  
  // Load IronCalc formula engine
  const formulaPlugin = await DataPrismPlugins.createProcessingPlugin('ironcalc-formula');
  await manager.loadPlugin(formulaPlugin);
  
  // Load LangGraph workflow orchestration
  const workflowPlugin = await DataPrismPlugins.createIntegrationPlugin('langgraph-integration');
  await manager.loadPlugin(workflowPlugin);
  
  // Execute formula
  const result = await manager.executePlugin('ironcalc-formula', {
    formula: 'SUM(A1:A10)',
    data: { A1: 10, A2: 20, A3: 30, A4: 40, A5: 50, A6: 60, A7: 70, A8: 80, A9: 90, A10: 100 }
  });
  
  console.log('Formula result:', result); // 550
</script>

Framework Only

For custom implementations, use the lightweight framework bundle and load plugins separately.

<!-- Framework Only -->
<script src="https://cdn.jsdelivr.net/npm/@dataprism/plugins@latest/dist/framework-bundles/dataprism-framework.min.js"></script>

<script>
  const manager = new DataPrismFramework.PluginManager();
  await manager.initialize();
  
  // Load specific plugins as needed
  // Plugins must be loaded separately when using framework-only bundle
</script>

NPM Installation

For Node.js projects and build systems, install via NPM:

# Install the complete package
npm install @dataprism/plugins

# Or install specific plugins
npm install @dataprism/plugins/ironcalc-formula
npm install @dataprism/plugins/langgraph-integration
// TypeScript/ES6 imports
import { PluginManager } from '@dataprism/plugins';
import { IronCalcFormulaPlugin } from '@dataprism/plugins/ironcalc-formula';
import { LangGraphIntegrationPlugin } from '@dataprism/plugins/langgraph-integration';

// Initialize plugins
const manager = new PluginManager();
await manager.loadPlugin(new IronCalcFormulaPlugin());
await manager.loadPlugin(new LangGraphIntegrationPlugin());

// Create and execute a workflow
const workflow = {
  id: 'excel-analysis',
  name: 'Excel-style Data Analysis',
  nodes: [
    {
      id: 'calculate',
      type: 'agent',
      agentId: 'formula-agent',
      configuration: { formulas: ['SUM(A:A)', 'AVERAGE(B:B)', 'MAX(C:C)'] }
    },
    {
      id: 'summarize',
      type: 'agent', 
      agentId: 'insight-agent',
      configuration: { includeCharts: true }
    }
  ],
  edges: [{ from: 'calculate', to: 'summarize' }],
  entryPoint: 'calculate'
};

const workflowResult = await manager.executePlugin('langgraph-integration', {
  operation: 'execute-workflow',
  workflowId: 'excel-analysis',
  input: { dataset: myData }
});
